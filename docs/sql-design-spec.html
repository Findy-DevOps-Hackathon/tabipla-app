2. coupons テーブル

2-1. CREATE TABLE

-- クーポンテーブル: 自治体が作成、観光者が閲覧・利用
CREATE TABLE coupons (
    id               TEXT PRIMARY KEY,         -- UUID v4（アプリ側で採番）
    spot_id          TEXT NOT NULL,              -- 対象スポット
    title            TEXT NOT NULL,              -- 表示名（例: ドリンク10%OFF）
    description      TEXT,                       -- 利用条件・備考（任意）
    discount_percent INTEGER NOT NULL,           -- 割引率 %（5〜10 を想定）
    created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- FK: スポット削除時にクーポンも連動削除
ALTER TABLE coupons
  ADD CONSTRAINT coupons_spot_id_spots_id_fk
  FOREIGN KEY (spot_id) REFERENCES spots(id) ON DELETE CASCADE;

-- INDEX: スポット別クーポン取得の高速化
CREATE INDEX coupons_spot_id_idx ON coupons USING btree (spot_id);

2-2. カラム定義と設計理由

カラム型制約設計理由
idTEXTPK UUID v4 をアプリ側で生成。spotsテーブルと同じ方式で統一。SERIAL を使わないのは、将来的にマルチDB/マイクロサービス環境でも衝突しないため
spot_idTEXTNOT NULL, FK → spots(id) CASCADE スポットに1:N で紐づける。1つのスポットに複数クーポンを発行可能（例: ドリンク割引＋入場割引）。CASCADE は「スポット自体が消えたらクーポンも意味を失う」ため
titleTEXTNOT NULL カード上で視認する表示名。最大128文字（API側バリデーション）
descriptionTEXTNULL許容 利用条件や備考。必須にしないのは「5%OFF」だけでも成立するケースがあるため
discount_percentINTEGERNOT NULL 割引率を%整数で保持。金額（円）でなく率にした理由 → 店舗ごとに価格帯が異なり、固定額では一律運用できない。5〜10%は下記の根拠による
created_atTIMESTAMPTZNOT NULL DEFAULT NOW() 作成日時。タイムゾーン付きで国際化対応
updated_atTIMESTAMPTZNOT NULL DEFAULT NOW() 更新日時。現状UPDATEはないが、将来の編集機能に備える

2-3. discount_percent = 5〜10% の根拠

なぜ 5〜10% なのか？ — 常設型・店舗負担型クーポンとして現実的な数字を業界事例から逆算
観点根拠
飲食の原価率 一般的な飲食店の原価率は 30〜35%、粗利 65〜70%。10%OFF でも粗利 55〜60% を確保でき、店舗側が持続可能
物販（お土産）の原価率 物販は原価率 40〜60% と幅が広い。原価率が高い商品では 10% が限度、5% が安全圏
入園料・施設 公共施設は団体割引 10% が慣例（例: 国立博物館の団体料金）。5〜10% はこの範囲内
自治体クーポン事例 GoToトラベル地域クーポンや、各自治体の「まちなかクーポン」は 5〜20%。ただし財源補填付き。自主財源なしの常設型なら 5〜10% が上限
リピート効果 20%以上は「お得感」で1回来訪、5〜10%は「いつでも使える安心感」で再訪を促進。旅プラの設計思想（何度でも使える＝消し込みなし）と整合
API バリデーション: 現在は minimum: 1, maximum: 100（schemas.ts）。チームで合意後、minimum: 5, maximum: 10 に絞ることを推奨

2-4. 持たないカラムとその理由

持たないカラム理由
use_count（利用回数） 「何度でも使える」設計。消し込み管理（使用済みフラグ/回数上限）は運用コストが高く、自治体の小規模運用に不向き
valid_from / valid_until（有効期間） 常設クーポンの想定。期間限定にしたい場合は削除で対応。期間カラムを持つとフロント側に「期限切れ表示」ロジックが必要になり複雑化する
discount_amount（割引金額） 率（%）で統一。店舗ごとに価格帯が異なるため、固定金額は一律運用できない
municipality_id（自治体ID） 現状は小諸市1自治体。spot_id → spots で自治体は暗黙的に特定できる。マルチ自治体化時に spots に municipality_id を追加すれば足りる
image_url（画像） MVP ではテキスト表示のみ。画像ストレージの設計は別タスク

3. recommendations テーブル

3-1. CREATE TABLE

-- おすすめ店テーブル: 自治体がキュレーションしたお食事処/お土産どころ
CREATE TABLE recommendations (
    id         TEXT PRIMARY KEY,              -- UUID v4
    spot_id    TEXT NOT NULL,                   -- 紐づく観光スポット
    type       TEXT NOT NULL,                   -- 'お食事処' | 'お土産'
    name       TEXT NOT NULL,                   -- 店名
    address    TEXT,                            -- 住所（Place lookup で自動入力）
    lat        DOUBLE PRECISION,               -- 緯度（Place lookup で自動入力）
    lon        DOUBLE PRECISION,               -- 経度（Place lookup で自動入力）
    comment    TEXT,                            -- 自治体のおすすめコメント
    url        TEXT,                            -- 参考URL（食べログ等）
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- FK: スポット削除時に連動削除
ALTER TABLE recommendations
  ADD CONSTRAINT recommendations_spot_id_spots_id_fk
  FOREIGN KEY (spot_id) REFERENCES spots(id) ON DELETE CASCADE;

-- INDEX: スポット別おすすめ店取得の高速化
CREATE INDEX recommendations_spot_id_idx ON recommendations USING btree (spot_id);

3-2. カラム定義と設計理由

カラム型制約設計理由
idTEXTPK UUID v4。coupons/spots と同じ方式
spot_idTEXTNOT NULL, FK → spots(id) CASCADE 「この観光スポット周辺のおすすめ」として紐づける。1スポットに複数の店を紐付可。CASCADE は coupons と同じ理由
typeTEXTNOT NULL 'お食事処' | 'お土産' の2値。ENUM型ではなく TEXT にしたのは、PostgreSQL の ENUM は ALTER TYPE が面倒（カラム追加に DDL が要る）で、将来「体験」「宿泊」等を追加するときに柔軟。API側で enum バリデーション
nameTEXTNOT NULL 店名。管理画面の店名検索（Place lookup）で自動入力。最大128文字（API側）
addressTEXTNULL許容 住所。Place lookup で自動入力されるが、手入力の場合は省略可
lat / lonDOUBLE PRECISIONNULL許容 緯度経度。将来の地図表示（ピン打ち）に備える。spots テーブルと同じ型。NULL許容は Place lookup が失敗した場合の手入力に対応
commentTEXTNULL許容 自治体の一言コメント（例:「小諸名物のくるみそば」）。ユーザー側では tooltip で表示
urlTEXTNULL許容 食べログ/ホットペッパー/公式サイト等のリンク。ユーザー側では店名クリックで外部遷移
created_at / updated_atTIMESTAMPTZNOT NULL DEFAULT NOW() coupons と同じ設計

3-3. type を TEXT にした理由（ENUM vs TEXT）

ENUM を避けた理由:
観点ENUMTEXT + API バリデーション（採用）
値の追加 ALTER TYPE ... ADD VALUE が必要。トランザクション内で使えない制約あり API の enum 配列に追加するだけ。DB マイグレーション不要
整合性 DB レベルで保証 API 入力時に enum: ["お食事処", "お土産"] で検証。不正値は 400 で弾く
将来拡張 DDL 変更 + マイグレーション コード変更のみ（例: "体験", "宿泊" を追加）

4. 共通設計方針

方針内容理由
ID 生成 UUID v4 を アプリ側（Node.js crypto.randomUUID） で採番 spots テーブルの既存方式に統一。DB の uuid-ossp 拡張に依存しない
FK + CASCADE spot_id → spots(id) ON DELETE CASCADE スポットが消えたらクーポン/おすすめ店も意味を失う。孤児レコード防止
INDEX spot_id に btree INDEX 「このスポットの〇〇一覧」が主クエリ。スポット数が増えても WHERE spot_id = ? を高速に保つ
ES 連携なし クーポン/おすすめ店は DB のみで完結 全文検索・ベクトル検索の対象ではない。ES に投入する意味がなく、reindex の複雑さを回避
タイムゾーン TIMESTAMPTZ（timestamp with time zone） spots / admin_users と統一。JSTだけでなく、UTC変換を安全に行える
Drizzle ORM schema.ts に pgTable 定義 → 型自動生成 TypeScript の型安全。$inferSelect / $inferInsert でリポジトリ層の型を自動導出

6. API エンドポイント一覧

coupons

メソッドパス認証用途
GET/coupons?spotId=...不要（公開）観光者がスポットのクーポン一覧を取得

メソッドパス認証用途
GET/recommendations?spotId=...不要（公開）観光者がスポット近隣のおすすめ店を取得

POST/admin/recommendations管理者トークン自治体がおすすめ店を登録（店名検索→自動入力）
DELETE/admin/recommendations/:id管理者トークン自治体がおすすめ店を削除

9. 既存テーブルとの整合性

既存テーブル新テーブルとの関係影響
spots coupons.spot_id, recommendations.spot_id の参照先 spots の DELETE 時に CASCADE で連動削除。spots の構造変更は不要
admin_users 直接の参照なし クーポン/おすすめ店の作成者カラムは持たない（1自治体運用のため）。マルチ自治体化時に created_by を追加すれば対応可
